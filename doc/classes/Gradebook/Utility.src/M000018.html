        def self.get_cell_feed(sps_client,sps_id,params='')
            cell_feed=Gradebook::Cache.cache_get_request(sps_client,'cell_feed',"https://spreadsheets.google.com/feeds/cells/#{sps_id}/1/private/full?prettyprint=true#{params}")
            return cell_feed
        end
