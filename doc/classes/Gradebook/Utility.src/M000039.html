        def self.sps_get_etag(sps_client,id)
                etag=nil
                @sps_feed=sps_client.get("https://spreadsheets.google.com/feeds/worksheets/#{id}/private/full").to_xml
                @sps_feed.elements.each('entry') do |entry|
                    etag=entry.attribute('etag').value
                end

            return etag
        end
