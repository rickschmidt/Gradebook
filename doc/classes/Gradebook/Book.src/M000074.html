       def create(doc_client,course_name)
           @doc_client=doc_client
            body="<?xml version='1.0' encoding='UTF-8'?>\n<entry xmlns=\"http://www.w3.org/2005/Atom\">\n<category scheme=\"http://schemas.google.com/g/2005#kind\"\nterm=\"http://schemas.google.com/docs/2007#spreadsheet\"/>\n<title>\#{course_name}</title>\n</entry>\n"
           @doc_feed=@doc_client.post('http://docs.google.com/feeds/documents/private/full',body) 
           puts @doc_feed
           
        end
