        def remove_category(category)
           response=Utility.remove_category(@client.sps_client,@sps_id,category) 
           if response==nil
               puts "\e[1;37mCategory does not exist.\e[0m"
           else
               puts "\e[1;34mCategory removed.\e[0m"
           end
        end
